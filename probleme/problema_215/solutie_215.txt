#include <iostream>

using namespace std;

int n,a[100][100],m;

int main(){
    cin >> m;
    n=6;
    int i=1;
    while(m){
        for(int j=i+1;j<=n;++j)
            a[i][j] = a[j][i] = m % 10;
        m/=10;
        i++;
    }
    for(int i=1;i<=n;++i){
        for(int j=1;j<=n;++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
