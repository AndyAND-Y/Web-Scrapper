#include<iostream>
#include<iomanip>

using namespace std;


int main()
{
    int n,d,s,k;
    cin>>n;
    s=1+n;k=2;
    for(d=2;d<=n/2;d++)
    {
        if(n%d==0)
        {
            s=s+d;
            k++;
        }
    }
    cout<<fixed<<setprecision(2)<<(float)s/k;
}
