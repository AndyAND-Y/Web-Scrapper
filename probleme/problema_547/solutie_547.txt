#include <iostream>
#include <cassert>
using namespace std;

int main(){
    int n, v[1005];
    
    cin  >> n;
    for(int i = 0 ; i < n ; ++ i)
        cin >> v[i];
    
    int Min = v[0], Max =v[0];
    for(int i = 0 ; i < n ; ++ i)
    {
        if(v[i] < Min)
            Min = v[i];
        if(v[i] > Max)
            Max = v[i];
    }
    int cnt = 0;
    for(int i =0 ; i < n; ++i)
        if(v[i] == Max - Min)
            cnt ++;
    cout << cnt;
    return 0;
}
