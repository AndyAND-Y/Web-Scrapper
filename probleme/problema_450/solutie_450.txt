#include <iostream>
using namespace std;

int a, b, c;

int main(){
    cin>>a>>b>>c;
    if (c == 1)
        cout<<a+b;
    if (c == 2)
        cout<<a-b;
    if (c == 3)
        cout<<a*b;
    if (c == 4)
        cout<<a/b;
    if (c == 5)
        cout<<a%b;
    return 0;
}
