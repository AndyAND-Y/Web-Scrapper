#include <iostream>
using namespace std;

int a[16][16], n;

int main(){
    cin  >> n;
    
    for(int j =1 ; j <= n ; j++)
        a[1][j] = j;
    
    for(int i = 2 ; i <= n ; i ++)
    {
        for(int j = 1 ; j < n ; j ++)
            a[i][j] = a[i-1][j+1];
        a[i][n] = a[i-1][1];
    }
    
    for (int i = 1 ;i <= n ; ++i){
        for(int j = 1 ; j <= n ; ++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
