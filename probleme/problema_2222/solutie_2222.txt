#include <iostream>

using namespace std;

int main(){
    int n , k , p;
    cin >> n >> k;
    p = 0;
    while( n > 1)
    {
        p ++;
        n /= 10;
    }
    
    cout << 1;
    
    int i = 1;
    while(i <= k){
        int cp = p;
        while(cp != 0)
        {
            cout << 0;
            cp --;
        }
        i ++;
    }
    return 0;
}
