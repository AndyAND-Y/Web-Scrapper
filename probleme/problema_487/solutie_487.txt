#include <iostream>

using namespace std;

int n, v[205];

int main()
{
    cin >> n;
    for(int i = 0; i < n ;++i)
        cin >> v[i];
    int s = 0 , cnt = 0;
    for(int i = 0; i < n ;++i)
        s += v[i];
    for(int i = 0; i < n ;++i)
        if(v[i] * n > s)
            cnt ++;
    cout << cnt;
    return 0;
}
