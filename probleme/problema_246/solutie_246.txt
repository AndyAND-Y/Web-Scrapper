#include <iostream>

using namespace std;

int main(){
    int n , x , a=0 , b=0;
    cin >> n;
    for(int i = 1; i<=n; ++i){
        cin >> x;
        if(x%2==1)
            b = a, a = x;
    }
    if(b!=0)
        cout << b << " " << a;
    else
        cout << "Numere insuficiente";
    return 0;
}
