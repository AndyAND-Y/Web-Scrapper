#include <iostream>
using namespace std;

int n,m, a[105][105];

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1; j <= m ; ++j)
            cin >> a[i][j];
    
    int I = 0 , max = 0;
    for(int i = 1 ; i <= n ; ++i)
    {
        int s = 0;
        for (int j = 1 ; j <= m ; ++j)
            if(a[i][j] % 2 == 0)
                s += a[i][j];
        if(s > max)
            max = s , I = i;
    }
    
    cout << I << endl;
    return 0;
}
