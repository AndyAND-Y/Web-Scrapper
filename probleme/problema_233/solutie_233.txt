#include <iostream>

using namespace std;

int n,a[100][100], k,p;

int main(){
    cin >> n;
    n *= 2;
    p = 1;
    for(int i=1 ; i<=n ; i+=2)
        for(int j=1;j<=n;++j)
            a[i][j] = p, p+=2;
    p = 2;
    for(int i=2 ; i<=n ; i+=2)
        for(int j=n;j;--j)
            a[i][j] = p, p+=2;
    for(int i=1;i<=n;++i){
        for(int j=1;j<=n;++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
