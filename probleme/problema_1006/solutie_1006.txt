#include <iostream>
using namespace std;

int main()
{
    for(int i = 1 ; i <= 5 ; i ++)
        for(int j = i + 1 ; j <= 6 ; j ++)
            for(int k = j + 1 ; k <= 7 ; k ++)
                for(int p = k + 1 ; p <= 8 ; p ++)
                    for(int q = p + 1 ; q <= 9 ; q ++)
                        cout << i << j << k << p << q << p << k << j << i << "\n";
    return 0;
}
