#include <iostream>
#include <cassert>
using namespace std;

int main(){
    int n, v[1005];
    cin  >> n;
    
    for(int i = 1 ; i <= n ; ++ i)
        cin >> v[i];
    int imin = 1 , imax = 1;
    for(int i = 2 ; i <= n ; ++i)
    {
        if(v[i]<v[imin])
            imin = i;
        if(v[i] > v[imax])
            imax = i;
    }
    cout << imin << " " << imax;
    return 0;
}
