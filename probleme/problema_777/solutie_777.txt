#include <iostream>
using namespace std;

int n,m,a[101][101];

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= m ; ++j)
            cin >> a[i][j];
    
    int cnt = 0;
    for(int j = 1 ; j <= m ; j ++)
    {
        bool ok = true;
        for(int i = 1 ; i <= n ; ++i)
            for(int k = i + 1 ; k <= n ; k ++)
                if(a[i][j] == a[k][j])
                    ok = false;
        if(ok)
            cnt ++;
    }
    
    cout << cnt;
    
    return 0;
}
