#include <iostream>
using namespace std;

int n, x[1005];

int main(){
    cin >> n;
    for(int i = 0 ; i < n; ++i)
        cin >> x[i];
    int p = -1, q = -1;
    for(int i = 0 ; i < n ; i ++)
        if(x[i] %2 == 1)
        {
            if(p == -1)
                p = i;
            q = i;
        }
    int s = 0;
    for(int  i = p ; i <= q ; i ++)
        s += x[i];
    cout << s;
    return 0;
}
