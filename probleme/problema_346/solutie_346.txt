#include <iostream>
using namespace std;

int main(){
    int n,a, max, ap;
    cin >> n;
    cin >> a;
    max = a, ap = 1;
    for(int i=2;i<=n;++i){
        cin >> a;
        if(a>max)
            max = a, ap = 1;
        else
            if(max == a)
                ap++;
    }
    cout << max << " " << ap;
    return 0;
}
