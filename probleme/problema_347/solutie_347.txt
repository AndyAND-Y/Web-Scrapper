#include <iostream>
using namespace std;

int main(){
    int n,a, max, min;
    cin >> n;
    cin >> a;
    min = max = a;
    for(int i=2;i<=n;++i){
        cin >> a;
        if(a>max)
            max = a;
        if(a<min)
            min = a;
    }
    cout << max + min;
    return 0;
}
