#include <iostream>
#include <cassert>
using namespace std;

int main()
{
    int cnt = 0;
    int a , b;
    cin >> a >> b;
    while(a != 0 && b != 0)
    {
        while(a > 9)
            a /= 10;
        while(b > 9)
            b /= 10;
        if(a == b)
            cnt ++;
        cin >> a >> b;
    }
    cout << cnt << endl;
    return 0;
}
