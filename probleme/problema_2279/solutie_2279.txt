#include <stdio.h>
int t, n, cp, hp, p, i, rez;
int main() {
    scanf("%d", &t);
    while (t) {
        scanf("%d%d", &n, &p);
        cp = hp = 0;
        for (i=1; i<=n; i++) {
            scanf("%d", &rez);
            if (p/2 <= rez) cp++;
            else if (p/10 >= rez) hp++;
        }
        if (cp == 1 && hp == 2) printf("%s\n", "da");
        else printf("%s\n", "nu");
        t--;
    }
}
