#include <iostream>
#include <cassert>
using namespace std;

int v[105];

int main(){
    int n,x;
    cin >> n;
    for(int i=1;i<=n;++i){
        cin >> x;
        if(x>0 && x<=n)
            v[x] ++;
    }
    int p = 1;
    for(int i=1;i<=n;++i)
        if(v[i]!=1)
            p = 0;
    if(p)
        cout << "DA";
    else
        cout << "NU";
    return 0;
}
