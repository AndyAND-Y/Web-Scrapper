#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long n,x,cifu,cifp,nou=0,p=1,c;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>x;
        c=1,nou=0,p=1;
        while(x)
        {
            if(c==1)
                cifu=x%10,x/=10;
            if(x<10)
                cifp=x%10,x/=10;
            else
            {
            nou=nou*10+x%10;
            c++;
            x/=10;
            }
        }
        for(int j=1;j<=c-1;j++)
            p=p*10;
        nou=nou+(p*cifp);
        nou=nou*10+cifu;
        cout<<nou<<" ";
    }
    return 0;
}
