#include <iostream>

using namespace std;

int main()
{
    int n , m ,a[51][51];
    cin >> n >> m;
    for(int i = 1 ; i <= n ; i ++)
        for(int j = 1 ; j <= m ; j ++)
            cin >> a[i][j];
    
    int x = a[n][m];
    for(int i = 1 ; i <= n ; i ++)
        for(int j = 1 ; j <= m ; j ++)
            if(i == 1 || i == n || j == 1 || j == m)
                a[i][j] = x;
    for(int i = 1 ; i <= n ; i ++)
    {
        for(int j = 1 ; j <= m ; j ++)
            cout << a[i][j] << " ";
        cout << "\n";
    }
    return 0;
}
