#include <iostream>
#include <fstream>
#include <cassert>

using namespace std;

ifstream fin("unice.in");
ofstream fout("unice.out");

int n,v[100],x;

int main(){
    fin >> n;
    for( ; n ; --n){
        assert(fin >> x);
        assert(x>=0 && x<100);
        v[x] ++;
    }
    for(int i = 0;i<100;++i)
        if(v[i]==1)
            fout << i << " ";
    return 0;
}
