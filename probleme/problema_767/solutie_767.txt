#include <iostream>
using namespace std;

int n,m,k,a[101][101], v[101];

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= m ; ++j)
            cin >> a[i][j];
    int s = 0;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= m ; ++j)
            if(a[i][j] % 2 == 0)
                s += a[i][j];
    cout << s << endl;
    return 0;
}
