#include <iostream>

using namespace std;
int n, a, b, c, maxim, minim;

int main(){
    cin>>n;
    a = n%10;
    b = n/10%10;
    c = n/100%10;

    maxim = a;
    if (b > maxim)
        maxim = b;
    if (c > maxim)
        maxim = c;

    minim = a;
    if (b < minim)
        minim = b;
    if (c < minim)
        minim = c;

    cout<<a + b + c - maxim - minim;

    return 0;
}
