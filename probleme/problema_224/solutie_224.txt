#include <iostream>
#include <fstream>
using namespace std;

ifstream fin("rotire.in");
ofstream fout("rotire.out");

int a[105][105],n,m, b[105][105];

int main(){
    fin >> m >> n;
    for(int i=1;i<=m;++i)
        for(int j=1;j<=n;++j)
            fin >> a[i][j];
    for(int j=1;j<=n;++j)
        for(int i=1;i<=m;i++)
            b[n-j+1][i] = a[i][j];
    for(int i=1;i<=n;++i){
        for(int j=1;j<=m;++j)
            fout << b[i][j] << " ";
        fout << endl;
    }
    return 0;
}
