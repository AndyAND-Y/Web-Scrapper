#include <iostream>
using namespace std;

int n, v[1005];

int main()
{
    cin >> n;
    for(int i = 0; i < n ;++i)
        cin >> v[i];
    for(int i = 0, j = n-1 ; i <= j ; i ++, j--)
    {
        cout << v[i] << " ";
        if(i < j)
            cout << v[j] << " ";
    }
    return 0;
}
