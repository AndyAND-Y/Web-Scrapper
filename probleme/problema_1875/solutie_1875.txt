#include <iostream>
#include <fstream>
using namespace std;
ifstream f("platou1.in");
ofstream g("platou1.out");
int main()
{
    int xmin=10, k=0,lg=1,x,y;
    f>>x;
    while(f>>y)
        if(x==y)lg++;
        else
        {
            if(k<lg)
            {
                k=lg;
                xmin=x;
            }
            else 
                if(k==lg && xmin>x) xmin=x;
            lg=1;
            x=y;
        }
    g<<k<<" "<<xmin<<endl;
    return 0;
}
