#include <iostream>
using namespace std;

int main(){
    int n , x , y , r , rez;
    
    cin >> n;
    
    rez  = n;
    
    for(int i  = 2 ; i< n; ++i){
        rez ++;
        for(int j = i+2 ; j <= n ; ++j)
        {
            x = j, y  = i;
            while(x % y != 0)
            {
                r = x % y;
                x = y;
                y = r;
            }
            if(y == 1)
                rez ++;
        }
    }
    cout << rez;
    return 0;
}
