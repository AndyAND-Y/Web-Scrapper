#include <iostream>
using namespace std;

int main(){
    int n;
    bool prim;
    cin >> n;
    cout << "2 ";
    for(int i = 3 ; i <= n ; i+=2){
        prim = true;
        for(int d = 3; d*d <= i; d += 2)
            if(i % d == 0)
                prim = false;
        if(prim)
            cout << i << " ";
    }
    return 0;
}
