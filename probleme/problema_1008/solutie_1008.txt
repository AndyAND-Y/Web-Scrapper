#include <iostream>
#include <fstream>
#include <cmath>
using namespace std;

ifstream fin ("spirala1.in");
ofstream fout("spirala1.out");

int n, a[101][101];

int main()
{
    fin >> n;   
    n = sqrt(n);
    int x;
    for(int k = 1 ; k <= n / 2 ; k ++)
    {
        for(int i = k ; i <= n - k + 1; i++)
        {
            fin >> x;
            a[i][k] = x;
        }
        for(int j = k + 1 ; j <= n - k + 1; j++)
        {
            fin >> x;
            a[n - k  + 1][j] = x;
        }
        for(int i = n - k ; i >= k ; i--)
        {
            fin >> x;
            a[i][n- k + 1] = x;
        }
        for(int j = n - k ; j > k; j--)
        {
            fin >> x;
            a[k][j] = x;
        }
    }
    if(n % 2 == 1)
    {
        fin >> x;
        a[n / 2 + 1][n / 2 + 1] = x;
    }
    
    for(int i = 1 ; i <= n ; i ++)
    {
        for(int j = 1 ; j <= n ; j ++)
            fout << a[i][j] << " ";
        fout << endl;
    }
    return 0;
}
