#include <iostream>

using namespace std;

unsigned long long n,m,nc;

int main(){
    cin>>n>>m;
    cout<<min(n,m)<<' ';
    if(n==m){
        cout<<1;
        return 0;
    }
    while(n>1||m>1){
        if(n>m){
            nc+=n/m;
            n-=m*(n/m);
        }
        else {
            nc+=m/n;
            m-=n*(m/n);
        }
    }
    cout<<nc;
    return 0;
}
