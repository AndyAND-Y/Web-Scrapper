#include <iostream>
#include <stdlib.h>

using namespace std;
int n;
int main()
{
    cin>>n;
    int n1, n2, p1, p2;
    n1=0;
    n2=0;
    p1=1;
    p2=1;
    while(n)
    {
        if(n%2==0)
        {
            n1=n1+n%10*p1;
            p1=p1*10;
        }
        else
        {
            n2=n2+n%10*p2;
            p2=p2*10;
        }
        n=n/10;
    }
    cout << abs(n1-n2);
    return 0;
}
