#include <iostream>

using namespace std;

int main()
{
    int n , up, x, au_fost = 0;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>x;
        if (x%2==0)
            up=x, au_fost = 1;
    }
    if(au_fost == 1)
        cout<<up;
    else
        cout << "IMPOSIBIL";
    return 0;
}
