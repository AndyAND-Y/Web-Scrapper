#include <iostream>
using namespace std;

int n,m,k,a[101][101], v[101];

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= m ; ++j)
            cin >> a[i][j];
    
    for(int i = 1 ; i <= n ; ++i)
        v[i] = a[i][1];
    for(int i =1 ; i <= n ; ++i)
        for(int j = 1 ; j < m ; j++)
            a[i][j] = a[i][j+1];
    for(int i = 1 ; i <= n ; ++i)
        a[i][m] = v[i];
    
    for(int i = 1 ; i <= n ; ++i)
    {
        for(int j = 1 ; j <= m ; ++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
