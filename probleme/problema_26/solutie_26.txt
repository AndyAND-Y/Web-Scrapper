int cmmnr(int n){
    int v[10]={0};
    while(n){
        v[n%10]++;
        n /= 10;
    }
    n = 0;
    int k=1;
    while(k<10 && v[k]==0)
        ++k;
    if(k<10)
        n = k, v[k]--;
    for(int i=0;i<10;++i)
        for(int j=1;j<=v[i];++j)
            n = 10*n + i;
    return n;
}
