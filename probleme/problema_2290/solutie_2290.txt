#include <iostream>
#include <cstdlib>

using namespace std;

int main()
{
    int n , x , y , ogl, dmin = 1000000000, xmin;
    for(cin >> n ; n ; n --)
    {
        cin >> x;
        y = x;
        ogl = 0;
        while(y)
            ogl = 10 * ogl + y % 10, y /= 10;
        
        if(abs(x - ogl) < dmin)
            dmin = abs(x - ogl), xmin = x;
        else
            if(abs(x - ogl) == dmin)
                if(x < xmin)
                    xmin = x;
    }
    cout << xmin << "\n";
    return 0;
}
