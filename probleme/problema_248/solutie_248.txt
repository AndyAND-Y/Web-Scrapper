#include <iomanip>
#include <fstream>
#include <iostream>
#include <cassert>
using namespace std;

ifstream fin("pozitie.in");
ofstream fout("pozitie.out");

int a[10];

int main(){
    int n, x, a, p;
    fin >> n >> x;
    p = 1;
    for(int i=2 ; i<=n ; ++i){
        fin >> a;
        if(a<x)
            p++;
    }
    fout << p;
    return 0;
}
