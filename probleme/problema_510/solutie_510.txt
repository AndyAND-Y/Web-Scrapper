#include <iostream>
using namespace std;

int main(){
    int n, x[1005];
    cin  >> n;  
    for(int i = 0 ; i < n ; ++ i)
        cin >> x[i];
    bool sortat = false;
    while(! sortat)
    {
        sortat = true;
        for(int i = 0 ; i < n - 1 ; i++)
            if(x[i] > x[i+1])
            {
                int aux = x[i];
                x[i] = x[i+1];
                x[i+1] = aux;
                sortat = false;
            }
    }
    for(int i = 0 ; i < n ; ++ i)
    {
        bool prim = true;
        if(x[i] < 2)
            prim = false;
        if(x[i] % 2 == 0 && x[i] > 2)
            prim = false;
        for(int d = 3 ; d * d <= x[i] ; d += 2)
            if(x[i] % d == 0)
                prim = false;
        if(prim)
            cout << x[i] << " ";
    }
    return 0;
}
