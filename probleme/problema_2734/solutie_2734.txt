#include <bits/stdc++.h>
using namespace std;

int main()
{
    vector<int> v;
    int n,x;
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>x,v.push_back(x);
    for(int i=0;i<v.size()-1;i++)
        for(int j=i+1;j<v.size();j++)
            if(v[i]==v[j])
                v.erase(v.begin()+j),j--;
    for(int i=0;i<v.size();i++)
        cout<<v[i]<<" ";
    return 0;
}
