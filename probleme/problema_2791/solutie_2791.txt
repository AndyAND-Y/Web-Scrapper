#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    int n , A[100][100], V[100];
    cin >> n;
    for(int i = 0 ; i < n ; i ++)
        cin >> V[i];
    for(int j = 0 ; j < n ; j ++)
        for(int i = 0 ; i < n ; i ++)
            A[i][j] = (j % 2 == 0? V[n - i - 1]:V[i]);
    for(int i = 0 ; i < n ; i ++)
    {
        for(int j = 0 ; j < n ; j ++)
            cout << A[i][j] << " ";
        cout << endl;
    }
    return 0;
    
}
