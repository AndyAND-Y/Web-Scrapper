#include <iostream>
#include <fstream>
#include <cassert>
using namespace std;

ifstream fin("paresort.in");
ofstream fout("paresort.out");

int n,m, a[1001];

int main(){
    fin >> n;
    m = 0;
    int x;
    for(int i=1;i<=n;++i){
        fin >>x ;
        if(x%2==0)
            a[++m] = x;
    }
    for(int i=1;i<m;++i)
        for(int j=i+1 ; j<=m ; ++j)
            if(a[i]>a[j]){
                int aux = a[i];
                a[i] = a[j];
                a[j] = aux;
            }
    if(m==0)
        fout << "nu exista";
    else{
        fout << m << endl;
        for(int i=1;i<=m;++i)
            fout << a[i] << " ";
    }
    return 0;
}
