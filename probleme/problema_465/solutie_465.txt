#include <iostream>
#include <cmath>
using namespace std;

int main(){
    int a, b , i , cnt = 0;;
    cin >> a >> b;
    i = sqrt(a);
    if(i * i < a)
        i ++;
    for( ; i*i <= b ; i++)
    {
        int n = i*i, ogl = 0;
        while(n)
            ogl = 10 * ogl + n % 10, n /= 10;
        n = sqrt(ogl);
        if(n * n == ogl)
            cnt ++;
    }
    
    cout << cnt;
    return 0;
}
