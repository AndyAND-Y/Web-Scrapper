#include <iostream>
using namespace std;

int main(){
    int n, x , cmax =-1 , max=-1;
    cin >> n;
    for(int i=1 ; i<=n ; ++i){
        cin >> x;
        int a = x;
        while(a > 9)
            a /= 10;
        if(a>cmax)
            cmax = a, max = x;
        else
            if(a == cmax)
                if(x > max)
                    max = x;
    }
    cout << max;
    return 0;
}
