#include <iostream>
#include <algorithm>
using namespace std;

int n,m,a[102][102];
int dx[]={0,0,-1,1}, dy[]={1,-1,0,0};

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= m ; ++j)
            cin >> a[i][j];
    
    int cnt = 0;
    
    for(int i = 1 ; i <= n ; i ++ )
        a[i][0] = a[i][m+1] = 0;
    for(int j =1 ; j <= m ; j ++)
        a[0][j] = a[n+1][j] = 0;
    
    for(int i = 1 ; i <= n ; i ++ )
        for(int j =1 ; j <= m ; j ++)
        {
            bool ok = true;
            for(int k = 0 ; k < 4 ; k ++)
                if(a[i+dx[k]][j+dy[k]] % 2 != 0)
                    ok = false;
            if(ok)
                cnt ++;
        }
    
    cout << cnt;
    return 0;
}
