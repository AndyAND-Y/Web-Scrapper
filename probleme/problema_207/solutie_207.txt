#include <iostream>

using namespace std;

int n,a[100][100];

int main(){
    cin >> n;
    for(int i=1;i<=n;++i)
        for(int j=1;j<=n;++j)
            a[i][j] = (i!=j?n-j+1:0);
    for(int i=1;i<=n;++i){
        for(int j=1;j<=n;++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
