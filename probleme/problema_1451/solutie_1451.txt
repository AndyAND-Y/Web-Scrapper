#include <iostream>
using namespace std;
int a[31][31],p[100],n,t=0;
void citire()
{
    int i,j;
    cin>>n;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
             cin>>a[i][j];
}
int main()
{
    int i,k,j,s=0;
    citire();
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            {
              s=s+a[i][j];
              p[t]=s;
            }
    while(s>0)
    {
        s=p[t];
        for(i=1;i<n-1;i++)
            for(j=1;j<n-1;j++)
            if(a[i][j]==1)
            {
                k=0;
                if(a[i-1][j]==0) k++;
                if(a[i+1][j]==0) k++;
                if(a[i][j-1]==0) k++;
                if(a[i][j+1]==0) k++;
                if(k>=2)
                {
                    s--;
                    a[i][j]=2;
                }
            }
        t++;
        p[t]=s;
        for(i=1;i<n-1;i++)
            for(j=1;j<n-1;j++)
               if(a[i][j]==2) a[i][j]=0;
    }
    cout<<t<<"\n";
    for(i=0;i<t;i++) cout<<p[i]<<"\n";
    return 0;
}
