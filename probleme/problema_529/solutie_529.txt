#include <iostream>
using namespace std;

int n , v[10], a[10];

int main(){
    for(int i = 0 ; i <10 ; i ++)
        a[i] = i;
    cin  >> n;  
    for(int i = 1 ; i <= n ;  ++i)
    {
        int x;
        cin >> x;
        do
        {
            v[x % 10] ++;
            x /= 10;
        }
        while(x > 0);
    }
    for(int i =0; i < 9 ; i++)
        for(int j= i + 1; j < 10 ; j ++)
            if(v[i]>v[j])
            {
                int aux = v[i];
                v[i] = v[j];
                v[j] = aux;
                aux = a[i]; a[i] = a[j]; a[j] = aux;
            }
            else
                if(v[i] == v[j])
                    if(a[i] > a[j])
                    {
                        int aux = a[i]; 
                        a[i] = a[j]; 
                        a[j] = aux;
                    }
    for(int i = 0 ; i < 10 ; i ++)
        if(v[i] > 0)
            cout << a[i] << " ";
    return 0;
}
