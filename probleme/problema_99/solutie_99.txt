#include <fstream>
using namespace std;

ifstream fin("nraparitii.in");
ofstream fout("nraparitii.out");

int main(){
    int n,x[100];
    fin >> n ;
    for(int i=0; i<n ;++i)
        fin >> x[i];
    int cnt = 0;
    for(int i=0; i<n; ++i)
        if(x[i] == x[n-1])
            cnt ++;
    fout << cnt << endl;
    return 0;
}
