#include <iostream>
#include <algorithm>
using namespace std;

int n,a[202][202], v[1000];

int main()
{
    cin >> n;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= n ; ++j)
            cin >> a[i][j];
    
    for(int i = 1 ; i <= n ; i ++ )
        for(int j =1 ; j <= n ; j ++)
            if(i > j && i + j > n + 1)
                v[a[i][j]] ++;
    for(int i = 0 ; i < 1000; i ++)
        if(v[i] > 1)
            cout << i << " ";
    return 0;
}
