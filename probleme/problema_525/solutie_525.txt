#include <iostream>
using namespace std;

int v[1000];

int main(){
    int n;
    cin  >> n;  
    for(int i = 1 ; i <= n ;  ++i)
    {
        int x ;
        cin >> x;
        if(x > 99 && x < 1000)
            v[x] = 1;
    }
    int  a = 0 , b = 0;
    for(int x = 999 ; x > 99 && a == 0 ; x --)
        if(v[x] == 0)
            if(b == 0)
                b = x;
            else
                a = x;
    if(a != 0)
        cout << a << " " << b;
    else
        cout << "NU EXISTA";
    return 0;
}
