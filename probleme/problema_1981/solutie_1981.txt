#include "iostream"

using namespace std;

int main()
{
    short nr, n;
    cin>>n;
    for(unsigned long long i=0 ; i< n ; ++i)
    {
    unsigned long long N = i*(1ULL<<i)+1;
    unsigned long long P = 1, p=0, d=2;
    while (N%d==0)
    {
        N/=d;
        p++;
    }
    if (p!=0)
        P *= (p+1);

    for (d=3; d*d<=N; d+=2)
    {
        p=0;
        while (N%d==0)
        {
            p++;
            N/=d;
        }
        if (p!=0)
            P *= (p+1);
    }
    if (N>1)
        P*=2;
        cout<<P<<' ';
    }
}
