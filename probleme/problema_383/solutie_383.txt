#include <iostream>
#include <cassert>
using namespace std;

int main(){
    int n,k,v[200];
    cin >> k >> n;
    
    for(int i = 0 ; i < n ; ++ i)
        cin >> v[i];
    for(int i = 0 ; i < n ; ++ i)
    {
        int x = v[i] / k * k;
        int y = x + k;
        if(v[i] - x <= y - v[i])
            v[i] = x;
        else
            v[i] = y;
    }
    for(int i = n - 1 ; i >= 0 ; -- i)
        cout << v[i] << " ";
    return 0;
}
