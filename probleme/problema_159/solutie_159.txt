#include <iostream>
using namespace std;

int n,a[100];

int main(){
    cin >> n;
    for(int i=1; i<=n; ++i)
        cin >> a[i];
    for(int p=n;p>=1;p--)
        if(a[p]%2==0){
            for(int i=n;i>=p;--i)
                a[i+1] = a[i];
            n++;
            a[p+1] = 2*a[p];
        }
    for(int i=1;i<=n;++i)
        cout << a[i] << " ";
    return 0;
}
