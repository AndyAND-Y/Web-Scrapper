#include <iostream>

using namespace std;

int main()
{
    int n , A[21][21];
    cin >> n;
    for(int i = 1; i <= n ; i ++)
        cin >> A[1][i];
    for(int i = 2; i <= n ; i ++)
    {
        for(int j = 1; j < n ; j ++)
            A[i][j+1] = A[i-1][j];
        A[i][1] = A[i-1][n];
    }
    
    for(int i = 1 ; i <= n ; i ++)
    {
        for(int j = 1 ; j <= n ; j ++)
            cout << A[i][j] << " ";
        cout << endl;
    }
    
    return 0;
}
