#include <iostream>
using namespace std;

int main(){
    int n, x[505];
    cin  >> n;  
    for(int i = 0 ; i < n ; ++ i)
        cin >> x[i];
    bool ok = true;
    for(int i = 1 ; i < n ; i++ )
        if(x[0] != x[i])
            ok = false;
    if(ok)
        cout << "DA";
    else
        cout << "NU";
    return 0;
}
