#include <iostream>
#include <cassert>

using namespace std;

int n,m,a[501];

int main(){
    cin >> n;;
    for(int i=1 ; i<=n ; ++i)
        assert(cin >> a[i]);
    int pp = 1;
    for(int i=1 ; i<=n ; ++i)
        if(a[i]%2!=0)
            pp = 0;
    if(pp)
        cout << "DA";
    else
        cout << "NU";
    return 0;
}
