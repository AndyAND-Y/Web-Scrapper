#include <cmath>
#include <iostream>
using namespace std;

int n,a[1005];

int prim(int n){
    if(n<2)
        return 0;
    if(n==2)
        return 1;
    if(n%2==0)
        return 0;
    for(int i=3 ; i*i<=n ; i+=2)
        if(n%i==0)
            return 0;
    return 1;
}

int main(){
    cin >> n;
    for(int i=1; i<=n; ++i)
        cin >> a[i];
    for(int p = n ; p >= 1 ; p--)
        if( prim(a[p]) ){
            for(int i = p ; i<n ; ++i)
                a[i] = a[i+1];
            n--;
        }
    for(int i=1;i<=n;++i)
        cout << a[i] << " ";
    return 0;
}
