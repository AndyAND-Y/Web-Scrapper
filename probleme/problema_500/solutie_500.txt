#include <iostream>
using namespace std;

int main(){
    int n, x[505];
    cin  >> n;  
    for(int i = 0 ; i < n ; ++ i)
        cin >> x[i];
    bool ordonat = true;
    for(int i = 0 ; i < n-1 ; i++ )
        if(x[i] > x[i+1])
            ordonat = false;
    if(ordonat)
        cout << "DA";
    else
        cout << "NU";
    return 0;
}
