#include <iostream>
using namespace std;

int main(){
    int n , d;
    long long C = 0;
    cin >> n;
    for(d=1 ; d*d<n ; d++)
        if(n % d == 0){
            if(d%2==0)
                C ++;
            if(n/d%2==0)
                C ++;
        }
    if(d * d == n)
        if(d%2==0)
            C ++;
    cout << C;
    return 0;
}
