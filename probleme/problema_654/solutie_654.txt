#include <iostream>
using namespace std;
int main ()
{
    long long int n;
    int cnt = 1;
    cin >> n;
    while(n != 1)
    {
        cnt ++;
        if(n % 2 == 0)
            n /= 2;
        else
            n = 3 * n + 1;
        //cout << n << "   ";
    }
    cout << cnt;
    return 0;

}
