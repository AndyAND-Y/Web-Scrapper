#include <iostream>

using namespace std;

int n,a[100][100];

int main(){
    cin >> n;
    int x = 1, y = 1, z;
    a[1][1] = a[1][2] = 1;
    for(int j=3;j<=n;++j){
        a[1][j] = (z = (x+y)%10);
        x = y;
        y = z;
    }
    for(int i=2;i<=n;++i)
        for(int j=1;j<=n;++j){
            a[i][j] = (z = (x+y)%10);
            x = y;
            y = z;
        }
    for(int i=1;i<=n;++i){
        for(int j=1;j<=n;++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
