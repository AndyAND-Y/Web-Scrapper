#include <iostream>

using namespace std;

int n,a[100][100];

int main(){
    cin >> n;
    for(int j = n ; j >= 1 ; --j)
        for(int i = 1 ; i <= n ; ++i)
            if(i==1 || i==n || j==1 || j==n)
                a[i][j] = i + j;
            else
                a[i][j] = a[i-1][j+1] + a[i][j+1] + a[i+1][j+1];
    for(int i=1;i<=n;++i){
        for(int j=1;j<=n;++j)
            cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
