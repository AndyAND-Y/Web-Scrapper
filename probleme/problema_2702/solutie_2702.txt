#include <stdio.h>
#include <stdlib.h>

int numarPerechi(int n, int* v) {
    int i;
    int sosete[101] = {0};
    int rezultat = 0;
    for (i = 0; i < n; i++) {
        sosete[v[i]]++;
        if (sosete[v[i]] == 2) {
            rezultat++;
            sosete[v[i]] = 0;
        }
    }
    return rezultat;
}

int main()
{
    int n, *v, i;
    scanf("%d", &n);
    v = (int*)malloc(sizeof(int)*n);
    for (i = 0; i < n; i++)
        scanf("%d", &v[i]);
    printf("%d", numarPerechi(n, v));
    return 0;
}
