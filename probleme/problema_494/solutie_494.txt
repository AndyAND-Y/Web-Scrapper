#include <iostream>
#include <cassert>
using namespace std;

int main(){
    int n, x[205], y[205];
    cin  >> n;  
    for(int i = 0 ; i < n ; ++ i)
        cin >> x[i];
    int s = 0;
    for(int i = 0 ; i < n ; ++ i)
        s += x[i];
    for(int  i = 0; i < n ; ++i)
        y[i] = s - x[i];
    for(int  i = 0; i < n ; ++i)
        cout << y[i] << " ";
    return 0;
}
