#include <iostream>
using namespace std;

int n,v[1005];

int main()
{
    cin >> n;
    for(int i = 0 ; i < n ; ++i)
        cin >> v[i];
    int maxim = v[0],minim = v[0];
    for(int i = 1; i < n ; ++i)
    {
        if(v[i] < minim)
            minim = v[i];
        if(v[i]>maxim)
            maxim = v[i];
    }
    cout << minim << " " << maxim;
    return 0;
}
