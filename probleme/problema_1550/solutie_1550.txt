#include<iostream>

using namespace std;

int main()
{
    int v[101],n,fact=1,flag=0,i;
    cin>>n;
    for( i=1; i<=n; i++)
    {
        fact=fact*i;
    }
   for(i=1; i<=n; i++) cin>>v[i];
   for(i=1; i<=n; i++)
    {
        if(fact%v[i]==0) cout<<v[i]<<" ";
        flag++;
    }
    if(flag==0) cout<<"Nu exista!";
    return 0;
}
