#include <iostream>
using namespace std;

int n,m, a[105][105];

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1; j <= m ; ++j)
            cin >> a[i][j];
    
    for(int i = 1 ; i <= n ; ++i)
    {
        int s = 0;
        for(int j = 1 ; j <= m ; ++j)
            s += a[i][j];
        cout << s << " ";
    }
    return 0;
}
