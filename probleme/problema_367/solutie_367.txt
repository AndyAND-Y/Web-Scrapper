#include <iostream>
using namespace std;

int main(){
    int n , s=0 , r , p , x;
    cin >> n;
    for(int i=1 ; i<=n ; i++){
        cin >> x;
        r = 0, p = 1;
        int uc = x % 10;
        while(x){
            if(x%10 != uc)
                r += x % 10 * p , p *= 10;
            x /= 10;
        }
        s += r;
    }
    cout << s;
    return 0;
}
