#include <iostream>
using namespace std;

int main(){
    int n;
    cin >> n;
    int s = 0;
    int i = 1;
    while(s<=n){
        i++;
        s = i*(i-1)/2;
        int p = n-s;
        if(p>0 && p%i==0)
        {
            for(int j=0;j<i;++j)
                cout<<p/i+j<<" ";
            cout << endl;
        }
    }
    return 0;
}
